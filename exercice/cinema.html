prix_mineur = 6

prix_majeur = 9

prix_popcorn = 5


print("Bienvenue dans le cinéma !")

age = int(input("Quelle age avez vous ? :"))


if age < 18:

    mineur = input(f"Vu que vous êtes mineur ça vous fais {prix_mineur}€. Voulez vous du pop-corn à {prix_popcorn}€ ? (oui/non) : ")

    if mineur == "oui":

        prix_final_mineur = prix_mineur + prix_popcorn

        print(f"Ok, ça vous fais un totale de {prix_final_mineur}€.")

    elif mineur == "non":

        print(f"Ok, ça vous fais un totale de {prix_mineur}€.")

else:

    majeur = input(f"Vu que vous êtes majeur ça vous fais {prix_majeur}€. Voulez vous du pop-corn à {prix_popcorn}€ ? (oui/non) : ")

    if majeur == "oui":

        prix_final_majeur = prix_majeur + prix_popcorn

        print(f"Ok, ça vous fais un totale de {prix_final_majeur}€.")

    elif majeur == "non":

        print(f"Ok, ça vous fais un totale de {prix_majeur}€.")
