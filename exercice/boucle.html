count_progression = 0

phrase = input("Quelle est la phrase que vous voulez qui se repete ? :")

count = int(input("Combien de fois voulez vous qu'elle se repète ? :"))


if count > 500:

    print("Se chiffre est trop élevé !")

else:

    print(f"Ok, pour résumer la phrase qui va se répeter {count} est {phrase}.")

    for i in range(count):

        count_progression += 1

        print(f"{count_progression}. {phrase}")

        if count_progression == 100:

            print("Vous avez dépassez les 100 répétissions !")
